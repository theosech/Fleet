194  for(size_t i=1;i<this->pool.size()-1;i++) { // never adjust i=0 (T=1) or the max temperature

198  sw[i] += k(this->pool[i].samples, v, t0) * (swap_history[i].mean()-swap_history[i+1].mean());
199  }
200 
201  }
202

204  for(size_t i=1;i<this->pool.size();i++) {
205  swap_history[i].reset();

211  for(size_t i=1;i<this->pool.size()-1;i++) {
212  this->pool[i].temperature = this->pool[i-1].temperature + exp(sw[i]);
