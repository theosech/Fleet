183 inline double p_delete_append(const std::string& x, const std::string& y, const float log_alphabet) {
194  // all of these get precomputed at compile time
195  static const float log_add_p = log(add_p);
196  static const float log_del_p = log(del_p);
197  static const float log_1madd_p = log(1.0-add_p);
198  static const float log_1mdel_p = log(1.0-del_p);
199 
200 
201  // Well we can always delete the whole thing and add on the remainder
202  float lp = log_del_p*x.length() + // we don't add log_1mdel_p again here since we can't delete past the beginning
203  (log_add_p-log_alphabet)*y.length() + log_1madd_p;
204 
205  // now as long as they are equal, we can take only down that far if we want
206  // here we index over mi, the length of the string that so far is equal
207  for(size_t mi=1;mi<=std::min(x.length(),y.length());mi++){
208  if(x[mi-1] == y[mi-1]) {
209  lp = logplusexp(lp, log_del_p*(x.length()-mi) + log_1mdel_p +
210  (log_add_p-log_alphabet)*(y.length()-mi) + log_1madd_p);
211  }
212  else {
213  break;
214  }
215  }
216 
217  return lp;
218 }

379 double p_KashyapOommen1984_edit(const std::string x, const std::string y, const double perr, const size_t nalphabet) {
380  // From Kashyap & Oommen, 1984
381  // perr gets used here to define the probability of each of these operations
382  // BUT Note here that we use the logs of these
383 
384  const int m = x.length(); // these must be ints or some negatives below go bad
385  const int n = y.length();
386  const int T = std::max(m,n); // max insertions allowed
387  const int E = std::max(m,n); // max deletions allowed
388  const int R = std::min(m,n);
389 
390  const double lp_insert = -log(nalphabet); // P(y_t|lambda) - probability of generating a single character from nothing
391  const double lp_delete = log(perr); // P(phi|d_x) -- probability of deleting
392 
393  Vector3D<double> W(T+1,E+1,T+1);
394 
395  // set to 1 in log space
396  W(0,0,0) = 0.0;
397 
398  // indexes into y and x, and returns the swap probability
399  // when they are different. Also corrects so i,j are 0-indexed
400  // instead of 1-indexed, as they are below.
401  // Note that when we are correct, we sum up both ways of getting the answer
402  const auto Sab = [&](int i, int j) -> double {
403  return log(perr / nalphabet + (1.0-perr)*(y[i-1]==x[j-1] ? 1 : 0));
404  };
405 
406  for(int t=1;t<=T;t++)
407  W(t,0,0) = W(t-1,0,0) + lp_insert;
408 
409  for(int d=1;d<=E;d++)
410  W(0,d,0) = W(0,d-1,0) + lp_delete;
411 
412  for(int s=1;s<=R;s++)
413  W(0,0,s) = W(0,0,s-1) + Sab(s-1,s-1);
414 
415  for(int t=1;t<=T;t++)
416  for(int d=1;d<=E;d++)
417  W(t,d,0) = logplusexp(W(t-1,d,0)+lp_insert,
418  W(t,d-1,0)+lp_delete);
419 
420  for(int t=1;t<=T;t++)
421  for(int s=1;s<=n-t;s++)
422  W(t,0,s) = logplusexp(W(t-1,0,s)+lp_insert,
423  W(t,0,s-1)+Sab(s+t-1,s-1));
424 
425  for(int d=1;d<=E;d++)
426  for(int s=1;s<=m-d;s++)
427  W(0,d,s) = logplusexp(W(0,d-1,s)+lp_delete,
428  W(0,d,s-1)+Sab(s-1,s+d-1));
429 
430  for(int t=1;t<=T;t++)
431  for(int d=1;d<=E;d++)
432  for(int s=1;s<=std::min(n-t,m-d);s++)
433  W(t,d,s) = logplusexp(W(t-1,d,s)+lp_insert,
434  logplusexp(W(t,d-1,s)+lp_delete,
435  W(t,d,s-1)+Sab(t+s-1,d+s-1))); // shoot me in the face
436 
437 
438  // now we sum up
439  double lp_yGx = -infinity; // p(Y|X)
440  for(int t=std::max(0,n-m); t<=std::min(T,E+n-m);t++){
441  double qt = lpmf_geometric(t+1,1.0-perr); // probability of t insertions -- but must use t+1 to allow t=0 (geometric is defined on 1,2,3,...)
442  lp_yGx = logplusexp(lp_yGx, W(t,m-n+t,n-t) + qt + lfactorial(m)+lfactorial(t)-lfactorial(m+t));
443  }
444  return lp_yGx;
445 
446 }
