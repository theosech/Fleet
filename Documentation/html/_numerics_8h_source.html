87 T logplusexp(const T a, const T b) {
88  // An approximate logplusexp -- the check on z<-25 makes it much faster since it saves many log,exp operations
89  // It is easy to derive a good polynomial approximation that is a bit faster (using sollya) on [-25,0] but that appears
90  // not to be worth it at this point.
91 
92  if (a == -infinity) return b;
93  else if(b == -infinity) return a;
94 
95  T mx = std::max(a,b);
96 
97  // compute a bound here at such that log1p(exp(z)) doesn't affect the result much (up to 1e-6)
98  // for floats, this ends up being about -13.8
99  // TODO: I Wish we could make this constexpr, but the math functions are not constexpr. Static is fine though
100  // it should only be run once at the start.
101  static const float breakout = get_log1pexp_breakout_bound<T>(1e-6, +[](T x) -> T { return log1p(exp(x)); });
102 
103 
104  T z = std::min(a,b)-mx;
105  if(z < breakout) {
106  return mx; // save us from having to do anything
107  }
108  else {
109  return mx + log1p(exp(z)); // mx + log(exp(a-mx)+exp(b-mx));
110  }
111 }
